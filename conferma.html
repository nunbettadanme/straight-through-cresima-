    const canvas = document.getElementById('gameCanvas');
    const ctx = canvas.getContext('2d');

    // Variabili di gioco
    let player = {
      x: 50,
      y: canvas.height / 2,
      width: 50,
      height: 50,
      speed: 5
    };

    let bullets = [];
    let zombies = [];

    // Funzione per disegnare il giocatore
    function drawPlayer() {
      ctx.fillStyle = 'blue';
      ctx.fillRect(player.x, player.y, player.width, player.height);
    }

    // Funzione per muovere il giocatore
    function movePlayer(direction) {
      if (direction === 'up' && player.y > 0) {
        player.y -= player.speed;
      } else if (direction === 'down' && player.y < canvas.height - player.height) {
        player.y += player.speed;
      }
    }

    // Funzione per sparare un proiettile
    function shoot() {
      bullets.push({ x: player.x + player.width, y: player.y + player.height / 2 });
    }

    // Funzione per disegnare i proiettili
    function drawBullets() {
      ctx.fillStyle = 'red';
      bullets.forEach(bullet => {
        ctx.fillRect(bullet.x, bullet.y, 10, 5);
      });
    }

    // Funzione per muovere i proiettili
    function moveBullets() {
      bullets.forEach(bullet => {
        bullet.x += 5;
      });
      bullets = bullets.filter(bullet => bullet.x < canvas.width);
    }

    // Funzione per generare gli zombie
    function generateZombies() {
      if (Math.random() < 0.02) {
        zombies.push({ x: canvas.width, y: Math.random() * canvas.height });
      }
    }

    // Funzione per disegnare gli zombie
    function drawZombies() {
      ctx.fillStyle = 'green';
      zombies.forEach(zombie => {
        ctx.fillRect(zombie.x, zombie.y, 50, 50);
      });
    }

    // Funzione per muovere gli zombie
    function moveZombies() {
      zombies.forEach(zombie => {
        zombie.x -= 2;
      });
      zombies = zombies.filter(zombie => zombie.x > 0);
    }

    // Funzione per rilevare collisioni tra proiettili e zombie
    function detectCollisions() {
      bullets.forEach(bullet => {
        zombies.forEach(zombie => {
          if (bullet.x < zombie.x + 50 && bullet.x + 10 > zombie.x && bullet.y < zombie.y + 50 && bullet.y + 5 > zombie.y) {
            bullets.splice(bullets.indexOf(bullet), 1);
            zombies.splice(zombies.indexOf(zombie), 1);
          }
        });
      });
    }

    // Funzione di gioco principale
    function gameLoop() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      moveBullets();
      moveZombies();
      generateZombies();
      detectCollisions();

      drawPlayer();
      drawBullets();
      drawZombies();

      requestAnimationFrame(gameLoop);
    }

    // Aggiungi eventi per muovere il giocatore e sparare
